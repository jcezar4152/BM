create database BlackManba;
use BlackManba;

create table Usuario
(
idUser int not null auto_increment,
nomeUser varchar(40) not null,
idadeUser date not null,
LoginUser varchar(40) not null,
SenhaUser varchar(40) not null,
primary key(id_cli)
);

create table roles
(
idRole int not null auto_increment,
nomeRole varchar(20) not null,
DataRole date not null,
primary key(id_dep)
);
